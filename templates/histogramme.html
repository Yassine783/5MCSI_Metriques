google.charts.load('current', {packages: ['corechart', 'bar']}); google.charts.setOnLoadCallback(drawBasic); function drawBasic() { // Génération de données aléatoires pour la température pour les 16 prochains jours var temperatureData = generateRandomData(16); var data = new google.visualization.DataTable(); data.addColumn('string', 'Day'); data.addColumn('number', 'Temperature (°C)'); data.addRows(temperatureData); var options = { title: 'Évolution des températures de Paris pour les 16 prochains jours', hAxis: { title: 'Jours' }, vAxis: { title: 'Température (°C)' } }; var chart = new google.visualization.ColumnChart( document.getElementById('chart_div')); chart.draw(data, options); } // Fonction pour générer des données aléatoires pour la température function generateRandomData(numDays) { var data = []; var currentDate = new Date(); currentDate.setDate(currentDate.getDate() + 1); // Commence à partir de demain for (var i = 0; i < numDays; i++) { var day = new Date(currentDate); day.setDate(day.getDate() + i); var temperature = Math.floor(Math.random() * (30 - (-10) + 1)) - 10; // Valeurs entre -10°C et 30°C data.push([day.toLocaleDateString(), temperature]); } return data; }
